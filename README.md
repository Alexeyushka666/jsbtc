JavaScript Bitcoin library


### Build:
    npm install jsbtc.js
    npm run build:wasm:prebuild
    npm run build:wasm
    npm run build
    npm run build:web
 
### Use in browser:
    <script src="jsbtc.test.js"></script>
    <script> ...
    // inside async function 
    var jsbtc = await jsbtc.asyncInit();
    ... </script>
